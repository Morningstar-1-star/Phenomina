    const useDriveToken = () => {
        const [token, setTokenState] = React.useState(() => localStorage.getItem('nxs_gdrive_token')||null);
        const [expiry, setExpiry] = React.useState(() => parseInt(localStorage.getItem('nxs_gdrive_expiry')||0));
        const tokenRef = React.useRef(token);

        const save = (tk, expiresIn) => {
            const exp = Date.now() + (expiresIn||3600)*1000 - 60000;
            localStorage.setItem('nxs_gdrive_token', tk);
            localStorage.setItem('nxs_gdrive_expiry', String(exp));
            tokenRef.current = tk;
            setTokenState(tk);
            setExpiry(exp);
        };
        const clear = () => {
            localStorage.removeItem('nxs_gdrive_token');
            localStorage.removeItem('nxs_gdrive_expiry');
            tokenRef.current = null;
            setTokenState(null);
        };
        const isValid = token && Date.now() < expiry;
        return { token: isValid ? token : null, tokenRef, save, clear, expired: token && !isValid };
    };
